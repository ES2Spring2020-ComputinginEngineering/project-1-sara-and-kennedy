#T = 2*math.pi*(math.sqrt(L/9.81))

#IMPORT STATMENTS
import numpy as np 
import matplotlib.pyplot as plt


#VARIABLES
length = np.array([5, 10, 15, 20, 25])



#CUSTOM FUNCTIONS
def findPeriod(length):
    period = (2*np.pi*(np.sqrt(length/9.81)))
    return period



#MAIN SCRIPT
period = findPeriod(length)

plt.plot(length, period)

plt.title('Period Values vs. Length')
plt.ylabel('Period Values (s)')
plt.xlabel('Length (in.)')
